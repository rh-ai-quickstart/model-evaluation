1. Forward the UI service port to access the application:

     oc port-forward svc/{{ .Values.ui.name }} 8080:8080 -n {{ .Release.Namespace }}

2. Verify the API is healthy:

{{- if .Values.routes.enabled }}
     curl -k https://{{ if .Values.routes.sharedHost }}{{ .Values.routes.sharedHost }}{{ else if .Values.routes.api.host }}{{ .Values.routes.api.host }}{{ else if .Values.routes.ui.host }}{{ .Values.routes.ui.host }}{{ else }}{{ .Values.api.name }}-{{ .Release.Namespace }}.apps.<your-cluster-domain>{{ end }}/health/
{{- else }}
     oc port-forward svc/{{ .Values.api.name }} 8000:8000 -n {{ .Release.Namespace }}
     curl http://localhost:8000/health/

3. Check deployment status:

     oc get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}
